 .MODEL SMALL
.STACK 100H
.DATA

a DB ?
b DB ?   
c DB ?

  
x DB 10,13,"Enter three Numbers : $"
y DB 10,13,"Largest Number : $"

ENDS   

.CODE

PROC MAIN
  
MOV AX,DATA
MOV DS,AX

LEA DX,x
MOV AH,9
INT 21H
MOV AH,1
INT 21H
MOV a,AL


MOV AH,1
INT 21H
MOV b,AL


MOV AH,1
INT 21H
MOV c,AL

LEA DX,y
MOV AH,9
INT 21H   
  
MOV AL,a
CMP AL,b
JNG DISPLAY2
CMP AL,c
JNG DISPLAY2
DISPLAY1:
MOV AH,2
MOV DL,a
INT 21H
JMP EXIT


DISPLAY2:
MOV BL,b
CMP BL,c
JNG DISPLAY3
MOV AH,2
MOV DL,b
INT 21H
JMP EXIT


DISPLAY3:
MOV AH,2
MOV DL,c
INT 21H


EXIT:
;EXIT TO DOS
MOV AX,4c00H
INT 21H
END MAIN
