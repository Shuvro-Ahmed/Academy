.MODEL SMALL
 
.STACK 100H

.DATA 

;variables
   
x db "Weekday$"
y db "Weekend"

.CODE 
MAIN PROC 

;initialize DS

MOV AX,@DATA 
MOV DS,AX 
 
; enter your code here

MOV AH,1
INT 21H

CMP AL,49
JE SAT

CMP AL,50
JE SUN 

CMP AL,51
JE MON 

CMP AL,52
JE TUE

CMP AL,53
JE WED

CMP AL,54
JE THU

CMP AL,55
JE FRI

SAT:
MOV AH,2 
MOV DL,0DH
INT 21H
MOV DL,0AH 
INT 21H

LEA DX,y
MOV AH,9
INT 21H
JMP EXIT

SUN:

MOV DL,0DH
INT 21H
MOV DL,0AH 
INT 21H

LEA DX,y

INT 21H
JMP EXIT

MON:

MOV DL,0DH
INT 21H
MOV DL,0AH 
INT 21H

LEA DX,x
MOV AH,9
INT 21H
JMP EXIT
 
TUE:
MOV AH,2 
MOV DL,0DH
INT 21H
MOV DL,0AH 
INT 21H

LEA DX,x
MOV AH,9
INT 21H
JMP EXIT

WED:
MOV AH,2 
MOV DL,0DH
INT 21H
MOV DL,0AH 
INT 21H

LEA DX,x
MOV AH,9
INT 21H
JMP EXIT  

THU:
MOV AH,2 
MOV DL,0DH
INT 21H
MOV DL,0AH 
INT 21H

LEA DX,x
MOV AH,9
INT 21H
JMP EXIT
        
FRI:
MOV AH,2 
MOV DL,0DH
INT 21H
MOV DL,0AH 
INT 21H

LEA DX,y
MOV AH,9
INT 21H
JMP EXIT        
        
;exit to DOS
 
EXIT:

MAIN ENDP
    END MAIN
